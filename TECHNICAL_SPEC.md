  Timer? _scrollTimer;
  Timer? _countdownTimer;
  bool _isPlaying = false;
  bool _isCountdown = false;

  PrompterSettings _settings = const PrompterSettings();
  
  // 60 FPS –ø—Ä–æ–∫—Ä—É—Ç–∫–∞ (16ms = 62.5 FPS)
  void _startScrolling() {
    _scrollTimer = Timer.periodic(
      const Duration(milliseconds: 16), (timer) {
        _scrollToNextPosition();
      });
  }
  
  // –ü–ª–∞–≤–Ω–∞ –ø—Ä–æ–∫—Ä—É—Ç–∫–∞ –±–µ–∑ –¥—å–æ—Ä–≥–∞–Ω–Ω—è

    _scrollController.jumpTo(newPosition); // jumpTo –¥–ª—è –ø–ª–∞–≤–Ω–æ—Å—Ç—ñ
  }
  
  // Countdown —Ñ—É–Ω–∫—Ü—ñ–æ–Ω–∞–ª
  void _startCountdown() {
    _isCountdown = true;
    _countdownValue = _settings.startDelay.ceil();
    _countdownTimer = Timer.periodic(
      const Duration(seconds: 1), (timer) {
        _countdownValue--;
        if (_countdownValue <= 0) {
          timer.cancel();
          _isCountdown = false;
          _startScrolling();
        }
        notifyListeners();
      });
  }
